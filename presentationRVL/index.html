Application
  • Defines the namespace
  • Manages other components
  • Controls the lifecycle
DEFINITION 
                            
    var App = Ember.Application.create();
INITIALIZATION 
                            
    App.ready = function() {
    /** INIT CODE GOES HERE **/
    }
    App.initialize();
Model
  • Represent objects
  • Define how data is stored
  • Contain no logic
  • Expanded by Ember Data
DEFINITION 
                            
    App.Movie = Ember.Object.extend({
        title : null,
        year: null,
        rating: 0,
        available: null,
        imdbLink: null
    });
USAGE 
                            
    var movie = App.Movie.create();
    movie.set('title','The Shawshank Redemption');
    movie.set('year', 1994);
    movie.set('rating', 9.2);
    movie.set('available', true);
    movie.set('imdbLink', 'http://www.imdb.com/title/tt0111161/');
VIEW
  • Encapsulates templates
  • Defines the UI
  • Handles user actions
DEFINITION 
                            
    App.MoviesView = Ember.View.extend({
        templateName: "movies",
        baseClass: "table",
        additionalClass: "table-hover"
    });
USAGE 
                            
    // JS
    App.moviesView = App.MoviesView.create().append();
    // HTML
    {{#view App.moviesView}}
    ...
    {{/view}}
    // HTML SELF CONTAINED
    {{view App.MoviesView}}
Controller
  • Contain the logic of the application
  • Manage object lists (Array Controller)
DEFINITION 
                            
    //Basic Controller
    App.ApplicationController = Ember.Controller.extend();

    //Array Controller
    App.MoviesController = Ember.ArrayController.extend({
        content : null,

        init: function(){
            this.content = []
        }
    });
USAGE 
                            
    App.moviesController = App.MoviesController.create();
Handlebars
  • Templating engine
  • Associated context
  • Reusable
  • Helpers
Usage 
                            
    //syntax
    {{App.myObject.property}}

    //iterate collections
    {{#each movie in App.moviesController}}
        {{movie.title}}
    {{/each}}

    //conditional
    {{#if App.myObject.property}}
        //...
    {{/if}}

    //client side actions
    <a href="#" {{action actionName parameter target="..."}} >Click me!</a>

ROUTER
  • Manages the state of the application
  • Initialize controllers and views
  • De-/Serialize urls
  • Injects views in templates
Usage 
                            
    App.Router = Ember.Router.extend({
        root: Ember.Route.extend({
            index: Ember.Route.extend({
                route: '/'
            }),
            movie: Ember.Route.extend({
                route: '/movie/:id',
                deserialize:  function(router, context){
                  var id = parseInt(context.id);
                  var controller = router.get('moviesController');
                  return controller.findProperty('id', id);
                },
                serialize:  function(router, context){
                  return {
                    id: context.id
                  }
                },
                connectOutlets: function(router, context){
                  //Set the movie in the controller
                  router.get('movieProfileController')
                                .set('content',context);
                  //Connect the outlet to the view/template/controller
                  router.get('applicationController')
                                .connectOutlet('movieProfile');
                }
            })
        })
    });

COMPUTED PROPERTIES
  • Composition of properties
  • Cached
  • Observers
  • Setters
Usage 
                            
    //Model computed property
    App.Movie = Ember.Object.extend({
        onSale: function(){
            return ((this.rating < 4 ) && this.available) ? true : false;
        }.property("rating", "available")
    });

    //Controller aggregated computed property
    App.MoviesController = Ember.ArrayController.extend({
        availableTitles: function(){
            return this.content.
                filterProperty('available',true).get('length');
        }.property('content.@each.available'),
    ...
    //Computed property with setter
    Person = Ember.Object.extend{
        //declared firstName and lastName as properties
        fullName: function(key,value){
            //getter
            if(arguments.length === 1) {
              return this.get('firstName') + ' ' + this.get('lastName');
            }
            //setter
            else {
                var name = value.split(" ");

                this.set('firstName', name[0]);
                this.set('lastName', name[1]);
            }
        }
    }
}
MIXINS
  • Encapsulates functionality
  • Applicable to any object
  • Reusable
Definition 
                            
    //Default Object
    App.Client = Ember.Object.extend(){
        name: null
        ...
    }

    //Our Mixins
    App.GeekMixin = Ember.Mixin.create(){
        favouriteMovie: null,
        init: function() {
            this._super();
            this.set("favouriteMovie", "Star Wars");
        }
    }

    App.HipsterMixin = Ember.Mixin.create(){
        favouriteDirector: null,
        init: function() {
            this._super();
            this.set("favouriteDirector", "Aronofsky");
        }
    }
Usage 
                            
    //Default initialization
    var clientA = App.Client.create(GeekMixin,HipsterMixin);
    clientA.set('name','Joe');

    clientA.get('favouriteMovie');      // Star Wars
    clientA.get('favouriteDirector');   // Aronofsky

    //Change default values
    var clientB = App.Client.create(GeekMixin,HipsterMixin, {
        init:function(){
        this._super();
        this.set('favouriteMovie', 'Star Trek');
        }
    });

    clientB.get('favouriteMovie'); //'Star Trek'
